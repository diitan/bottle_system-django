# ALLOWED_HOSTS
# - Nếu DEBUG thì cho dễ local
# - Nếu không, lấy từ env ALLOWED_HOSTS (chuỗi phân tách ,), lọc rỗng
_raw_allowed = os.getenv("ALLOWED_HOSTS", "").strip()
if DEBUG:
    #ALLOWED_HOSTS = ["*"]
    ALLOWED_HOSTS = ["127.0.0.1", "localhost"]
else:
    ALLOWED_HOSTS = [h.strip() for h in _raw_allowed.split(",") if h.strip()]

# Nếu deploy trên Render, có thể có RENDER_EXTERNAL_HOSTNAME -> thêm vào ALLOWED_HOSTS
RENDER_EXTERNAL_HOSTNAME = os.getenv("RENDER_EXTERNAL_HOSTNAME")
if RENDER_EXTERNAL_HOSTNAME:
    ALLOWED_HOSTS.append(RENDER_EXTERNAL_HOSTNAME)

# (Tuỳ chọn) cho phép mọi subdomain onrender.com khi cần
if not DEBUG:
    # chỉ thêm trên production nếu bạn muốn
    ALLOWED_HOSTS.append(".onrender.com")

# CSRF_TRUSTED_ORIGINS
# Lấy danh sách từ env CSRF_TRUSTED_ORIGINS nếu có
_raw_csrf = os.getenv("CSRF_TRUSTED_ORIGINS", "").strip()
_csrf_list = [s.strip() for s in _raw_csrf.split(",") if s.strip()]

# Chỉ giữ các origin hợp lệ (bắt đầu bằng http:// hoặc https://)
valid_csrf = [u for u in _csrf_list if u.startswith("http://") or u.startswith("https://")]

# Nếu đang phát triển local (DEBUG) và không có origin hợp lệ, đặt mặc định cho runserver
if DEBUG and not valid_csrf:
    valid_csrf = [
        "http://127.0.0.1:8000",
        "http://localhost:8000",
    ]

CSRF_TRUSTED_ORIGINS = valid_csrf
